323  public override void ReceiveSignal(Signal signal, Connection connection)
324  {
325  switch (connection.Name)
326  {
327  case "toggle":
328  if (signal.value != "0")
329  {
330  if (!IgnoreContinuousToggle || lastToggleSignalTime < Timing.TotalTime - 0.1)
331  {
332  IsOn = !IsOn;
333  }
334  lastToggleSignalTime = Timing.TotalTime;
335  }
336  break;
337  case "set_state":
338  IsOn = signal.value != "0";
339  break;
340  case "set_color":
341  if (signal.value != prevColorSignal)
342  {
343  LightColor = XMLExtensions.ParseColor(signal.value, false);
344 #if CLIENT
345  SetLightSourceState(Light.Enabled, currentBrightness);
346 #endif
347  prevColorSignal = signal.value;
348  }
349  break;
350  }
