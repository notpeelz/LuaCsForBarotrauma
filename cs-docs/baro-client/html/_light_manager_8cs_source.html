181  public void RenderLightMap(GraphicsDevice graphics, SpriteBatch spriteBatch, Camera cam, RenderTarget2D backgroundObstructor = null)
182  {
183  if (!LightingEnabled) { return; }
184 
185  if (Math.Abs(currLightMapScale - GameSettings.CurrentConfig.Graphics.LightMapScale) > 0.01f)
186  {
187  //lightmap scale has changed -> recreate render targets
188  CreateRenderTargets(graphics);
189  }
190 
191  Matrix spriteBatchTransform = cam.Transform * Matrix.CreateScale(new Vector3(GameSettings.CurrentConfig.Graphics.LightMapScale, GameSettings.CurrentConfig.Graphics.LightMapScale, 1.0f));
192  Matrix transform = cam.ShaderTransform
193  * Matrix.CreateOrthographic(GameMain.GraphicsWidth, GameMain.GraphicsHeight, -1, 1) * 0.5f;
194 
195  bool highlightsVisible = UpdateHighlights(graphics, spriteBatch, spriteBatchTransform, cam);
196 
197  Rectangle viewRect = cam.WorldView;
198  viewRect.Y -= cam.WorldView.Height;
199  //check which lights need to be drawn
200  recalculationCount = 0;
201  activeLights.Clear();
202  foreach (LightSource light in lights)
203  {
204  if (!light.Enabled) { continue; }
205  if ((light.Color.A < 1 || light.Range < 1.0f) && !light.LightSourceParams.OverrideLightSpriteAlpha.HasValue) { continue; }
206 
207  if (light.ParentBody != null)
208  {
209  light.ParentBody.UpdateDrawPosition();
210 
211  Vector2 pos = light.ParentBody.DrawPosition;
212  if (light.ParentSub != null) { pos -= light.ParentSub.DrawPosition; }
213  light.Position = pos;
214  }
215 
216  float range = light.LightSourceParams.TextureRange;
217  if (light.LightSprite != null)
218  {
219  float spriteRange = Math.Max(
220  light.LightSprite.size.X * light.SpriteScale.X * (0.5f + Math.Abs(light.LightSprite.RelativeOrigin.X - 0.5f)),
221  light.LightSprite.size.Y * light.SpriteScale.Y * (0.5f + Math.Abs(light.LightSprite.RelativeOrigin.Y - 0.5f)));
222 
223  float targetSize = Math.Max(light.LightTextureTargetSize.X, light.LightTextureTargetSize.Y);
224  range = Math.Max(Math.Max(spriteRange, targetSize), range);
225  }
226  if (!MathUtils.CircleIntersectsRectangle(light.WorldPosition, range, viewRect)) { continue; }
227 
228  light.Priority = lightPriority(range, light);
229 
230  int i = 0;
231  while (i < activeLights.Count && light.Priority < activeLights[i].Priority)
232  {
233  i++;
234  }
235  activeLights.Insert(i, light);
236  }
237  ActiveLightCount = activeLights.Count;
238 
239  float lightPriority(float range, LightSource light)
240  {
241  return
242  range *
243  ((Character.Controlled?.Submarine != null && light.ParentSub == Character.Controlled?.Submarine) ? 2.0f : 1.0f) *
244  (light.CastShadows ? 10.0f : 1.0f) *
245  (light.LightSourceParams.OverrideLightSpriteAlpha ?? (light.Color.A / 255.0f));
246  }
247 
248  //find the lights with an active light volume
249  activeLightsWithLightVolume.Clear();
250  foreach (var activeLight in activeLights)
251  {
252  if (activeLight.Range < 1.0f || activeLight.Color.A < 1 || activeLight.CurrentBrightness <= 0.0f) { continue; }
253  activeLightsWithLightVolume.Add(activeLight);
254  }
255 
256  //remove some lights with a light volume if there's too many of them
257  if (activeLightsWithLightVolume.Count > GameSettings.CurrentConfig.Graphics.VisibleLightLimit && Screen.Selected is { IsEditor: false })
258  {
259  for (int i = GameSettings.CurrentConfig.Graphics.VisibleLightLimit; i < activeLightsWithLightVolume.Count; i++)
260  {
261  activeLights.Remove(activeLightsWithLightVolume[i]);
262  }
263  }
264  activeLights.Sort((l1, l2) => l1.LastRecalculationTime.CompareTo(l2.LastRecalculationTime));
265 
266  //draw light sprites attached to characters
267  //render into a separate rendertarget using alpha blending (instead of on top of everything else with alpha blending)
268  //to prevent the lights from showing through other characters or other light sprites attached to the same character
269  //---------------------------------------------------------------------------------------------------
270  graphics.SetRenderTarget(LimbLightMap);
271  graphics.Clear(Color.Black);
272  graphics.BlendState = BlendState.NonPremultiplied;
273  spriteBatch.Begin(SpriteSortMode.BackToFront, BlendState.NonPremultiplied, transformMatrix: spriteBatchTransform);
274  foreach (LightSource light in activeLights)
275  {
276  if (light.IsBackground || light.CurrentBrightness <= 0.0f) { continue; }
277  //draw limb lights at this point, because they were skipped over previously to prevent them from being obstructed
278  if (light.ParentBody?.UserData is Limb limb && !limb.Hide) { light.DrawSprite(spriteBatch, cam); }
279  }
280  spriteBatch.End();
281 
282  //draw background lights
283  //---------------------------------------------------------------------------------------------------
284  graphics.SetRenderTarget(LightMap);
285  graphics.Clear(AmbientLight);
286  graphics.BlendState = BlendState.Additive;
287  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, transformMatrix: spriteBatchTransform);
288  Level.Loaded?.BackgroundCreatureManager?.DrawLights(spriteBatch, cam);
289  foreach (LightSource light in activeLights)
290  {
291  if (!light.IsBackground || light.CurrentBrightness <= 0.0f) { continue; }
292  light.DrawSprite(spriteBatch, cam);
293  light.DrawLightVolume(spriteBatch, lightEffect, transform, recalculationCount < MaxLightVolumeRecalculationsPerFrame, ref recalculationCount);
294  }
295  GameMain.ParticleManager.Draw(spriteBatch, true, null, Particles.ParticleBlendState.Additive);
296  spriteBatch.End();
297 
298  //draw a black rectangle on hulls to hide background lights behind subs
299  //---------------------------------------------------------------------------------------------------
300 
301  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Opaque, transformMatrix: spriteBatchTransform);
302  Dictionary<Hull, Rectangle> visibleHulls = GetVisibleHulls(cam);
303  foreach (KeyValuePair<Hull, Rectangle> hull in visibleHulls)
304  {
305  GUI.DrawRectangle(spriteBatch,
306  new Vector2(hull.Value.X, -hull.Value.Y),
307  new Vector2(hull.Value.Width, hull.Value.Height),
308  hull.Key.AmbientLight == Color.TransparentBlack ? Color.Black : hull.Key.AmbientLight.Multiply(hull.Key.AmbientLight.A / 255.0f), true);
309  }
310  spriteBatch.End();
311 
312  SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidColor"];
313  SolidColorEffect.Parameters["color"].SetValue(AmbientLight.Opaque().ToVector4());
314  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, transformMatrix: spriteBatchTransform, effect: SolidColorEffect);
315  Submarine.DrawDamageable(spriteBatch, null);
316  spriteBatch.End();
317 
318  graphics.BlendState = BlendState.Additive;
319 
320 
321  //draw the focused item and character to highlight them,
322  //and light sprites (done before drawing the actual light volumes so we can make characters obstruct the highlights and sprites)
323  //---------------------------------------------------------------------------------------------------
324  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, transformMatrix: spriteBatchTransform);
325  foreach (LightSource light in activeLights)
326  {
327  //don't draw limb lights at this point, they need to be drawn after lights have been obstructed by characters
328  if (light.IsBackground || light.ParentBody?.UserData is Limb || light.CurrentBrightness <= 0.0f) { continue; }
329  light.DrawSprite(spriteBatch, cam);
330  }
331  spriteBatch.End();
332 
333  if (highlightsVisible)
334  {
335  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive);
336  spriteBatch.Draw(HighlightMap, Vector2.Zero, Color.White);
337  spriteBatch.End();
338  }
339 
340  //draw characters to obstruct the highlighted items/characters and light sprites
341  //---------------------------------------------------------------------------------------------------
342  if (cam.Zoom > ObstructLightsBehindCharactersZoomThreshold)
343  {
344  SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidVertexColor"];
345  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
346  foreach (Character character in Character.CharacterList)
347  {
348  if (character.CurrentHull == null || !character.Enabled || !character.IsVisible) { continue; }
349  if (Character.Controlled?.FocusedCharacter == character) { continue; }
350  Color lightColor = character.CurrentHull.AmbientLight == Color.TransparentBlack ?
351  Color.Black :
352  character.CurrentHull.AmbientLight.Multiply(character.CurrentHull.AmbientLight.A / 255.0f).Opaque();
353  foreach (Limb limb in character.AnimController.Limbs)
354  {
355  if (limb.DeformSprite != null) { continue; }
356  limb.Draw(spriteBatch, cam, lightColor);
357  }
358  }
359  spriteBatch.End();
360 
361  DeformableSprite.Effect.CurrentTechnique = DeformableSprite.Effect.Techniques["DeformShaderSolidVertexColor"];
362  DeformableSprite.Effect.CurrentTechnique.Passes[0].Apply();
363  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied, transformMatrix: spriteBatchTransform);
364  foreach (Character character in Character.CharacterList)
365  {
366  if (character.CurrentHull == null || !character.Enabled || !character.IsVisible) { continue; }
367  if (Character.Controlled?.FocusedCharacter == character) { continue; }
368  Color lightColor = character.CurrentHull.AmbientLight == Color.TransparentBlack ?
369  Color.Black :
370  character.CurrentHull.AmbientLight.Multiply(character.CurrentHull.AmbientLight.A / 255.0f).Opaque();
371  foreach (Limb limb in character.AnimController.Limbs)
372  {
373  if (limb.DeformSprite == null) { continue; }
374  limb.Draw(spriteBatch, cam, lightColor);
375  }
376  }
377  spriteBatch.End();
378  }
379 
380  DeformableSprite.Effect.CurrentTechnique = DeformableSprite.Effect.Techniques["DeformShader"];
381  graphics.BlendState = BlendState.Additive;
382 
383  //draw the actual light volumes, additive particles, hull ambient lights and the halo around the player
384  //---------------------------------------------------------------------------------------------------
385  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, transformMatrix: spriteBatchTransform);
386 
387  spriteBatch.Draw(LimbLightMap, new Rectangle(cam.WorldView.X, -cam.WorldView.Y, cam.WorldView.Width, cam.WorldView.Height), Color.White);
388 
389  foreach (ElectricalDischarger discharger in ElectricalDischarger.List)
390  {
391  discharger.DrawElectricity(spriteBatch);
392  }
393 
394  foreach (LightSource light in activeLights)
395  {
396  if (light.IsBackground || light.CurrentBrightness <= 0.0f) { continue; }
397  light.DrawLightVolume(spriteBatch, lightEffect, transform, recalculationCount < MaxLightVolumeRecalculationsPerFrame, ref recalculationCount);
398  }
399 
400  lightEffect.World = transform;
401 
402  GameMain.ParticleManager.Draw(spriteBatch, false, null, Particles.ParticleBlendState.Additive);
403 
404  if (Character.Controlled != null)
405  {
406  DrawHalo(Character.Controlled);
407  }
408  else
409  {
410  foreach (Character character in Character.CharacterList)

446  private bool UpdateHighlights(GraphicsDevice graphics, SpriteBatch spriteBatch, Matrix spriteBatchTransform, Camera cam)
447  {
448  if (GUI.DisableItemHighlights) { return false; }
449 
450  highlightedEntities.Clear();
451  if (Character.Controlled != null && (!Character.Controlled.IsKeyDown(InputType.Aim) || Character.Controlled.HeldItems.Any(it => it.GetComponent<Sprayer>() == null)))
452  {
453  if (Character.Controlled.FocusedItem != null)
454  {
455  highlightedEntities.Add(Character.Controlled.FocusedItem);
456  }
457  if (Character.Controlled.FocusedCharacter != null)
458  {
459  highlightedEntities.Add(Character.Controlled.FocusedCharacter);
460  }
461  foreach (Item item in Item.ItemList)
462  {
463  if ((item.IsHighlighted || item.IconStyle != null) && !highlightedEntities.Contains(item))
464  {
465  highlightedEntities.Add(item);
466  }
467  }
468  }
469  if (highlightedEntities.Count == 0) { return false; }
470 
471  //draw characters in light blue first
472  graphics.SetRenderTarget(HighlightMap);
473  SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidColor"];
474  SolidColorEffect.Parameters["color"].SetValue(Color.LightBlue.ToVector4());
475  SolidColorEffect.CurrentTechnique.Passes[0].Apply();
476  DeformableSprite.Effect.CurrentTechnique = DeformableSprite.Effect.Techniques["DeformShaderSolidColor"];
477  DeformableSprite.Effect.Parameters["solidColor"].SetValue(Color.LightBlue.ToVector4());
478  DeformableSprite.Effect.CurrentTechnique.Passes[0].Apply();
479  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, samplerState: SamplerState.LinearWrap, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
480  foreach (Entity highlighted in highlightedEntities)
481  {
482  if (highlighted is Item item)
483  {
484  if (item.IconStyle != null && (item != Character.Controlled.FocusedItem || Character.Controlled.FocusedItem == null))
485  {
486  //wait until next pass
487  }
488  else
489  {
490  item.Draw(spriteBatch, false, true);
491  }
492  }
493  else if (highlighted is Character character)
494  {
495  character.Draw(spriteBatch, cam);
496  }
497  }
498  spriteBatch.End();
499 
500  //draw items with iconstyles in the style's color
501  spriteBatch.Begin(SpriteSortMode.Immediate, BlendState.Additive, samplerState: SamplerState.LinearWrap, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
502  foreach (Entity highlighted in highlightedEntities)
503  {
504  if (highlighted is Item item)
505  {
506  if (item.IconStyle != null && (item != Character.Controlled.FocusedItem || Character.Controlled.FocusedItem == null))
507  {
508  SolidColorEffect.Parameters["color"].SetValue(item.IconStyle.Color.ToVector4());
509  SolidColorEffect.CurrentTechnique.Passes[0].Apply();
510  item.Draw(spriteBatch, false, true);
511  }
512  }
513  }
514  spriteBatch.End();
515 
516  //draw characters in black with a bit of blur, leaving the white edges visible
517  float phase = (float)(Math.Sin(Timing.TotalTime * 3.0f) + 1.0f) / 2.0f; //phase oscillates between 0 and 1
518  Vector4 overlayColor = Color.Black.ToVector4() * MathHelper.Lerp(0.5f, 0.9f, phase);
519  SolidColorEffect.Parameters["color"].SetValue(overlayColor);
520  SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidColorBlur"];
521  SolidColorEffect.CurrentTechnique.Passes[0].Apply();
522  DeformableSprite.Effect.Parameters["solidColor"].SetValue(overlayColor);
523  DeformableSprite.Effect.CurrentTechnique.Passes[0].Apply();
524  spriteBatch.Begin(SpriteSortMode.Immediate, BlendState.NonPremultiplied, samplerState: SamplerState.LinearWrap, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
525  foreach (Entity highlighted in highlightedEntities)
526  {
527  if (highlighted is Item item)
528  {
529  SolidColorEffect.Parameters["blurDistance"].SetValue(0.02f);
530  item.Draw(spriteBatch, false, true);
531  }
532  else if (highlighted is Character character)
533  {
534  SolidColorEffect.Parameters["blurDistance"].SetValue(0.05f);
535  character.Draw(spriteBatch, cam);
536  }
537  }
538  spriteBatch.End();
539 
540  //raster pattern on top of everything
541  spriteBatch.Begin(blendState: BlendState.NonPremultiplied, samplerState: SamplerState.LinearWrap);
542  spriteBatch.Draw(highlightRaster,
543  new Rectangle(0, 0, HighlightMap.Width, HighlightMap.Height),
544  new Rectangle(0, 0, (int)(HighlightMap.Width / currLightMapScale * 0.5f), (int)(HighlightMap.Height / currLightMapScale * 0.5f)),
545  Color.White * 0.5f);
546  spriteBatch.End();
547 
548  DeformableSprite.Effect.CurrentTechnique = DeformableSprite.Effect.Techniques["DeformShader"];
549 
550  return true;
551  }

554  private Dictionary<Hull, Rectangle> GetVisibleHulls(Camera cam)
555  {
556  visibleHulls.Clear();
557  foreach (Hull hull in Hull.HullList)
558  {
559  if (hull.HiddenInGame) { continue; }
560  var drawRect =
561  hull.Submarine == null ?
562  hull.Rect :
563  new Rectangle((int)(hull.Submarine.DrawPosition.X + hull.Rect.X), (int)(hull.Submarine.DrawPosition.Y + hull.Rect.Y), hull.Rect.Width, hull.Rect.Height);
564 
565  if (drawRect.Right < cam.WorldView.X || drawRect.X > cam.WorldView.Right ||
566  drawRect.Y - drawRect.Height > cam.WorldView.Y || drawRect.Y < cam.WorldView.Y - cam.WorldView.Height)
567  {
568  continue;
569  }
570  visibleHulls.Add(hull, drawRect);
571  }
572  return visibleHulls;
573  }

575  public void UpdateObstructVision(GraphicsDevice graphics, SpriteBatch spriteBatch, Camera cam, Vector2 lookAtPosition)
576  {
577  if ((!LosEnabled || LosMode == LosMode.None) && !ObstructVision) return;
578  if (ViewTarget == null) return;
579 
580  graphics.SetRenderTarget(LosTexture);
581 
582  if (ObstructVision)
583  {
584  graphics.Clear(Color.Black);
585  Vector2 diff = lookAtPosition - ViewTarget.WorldPosition;
586  diff.Y = -diff.Y;
587  if (diff.LengthSquared() > 20.0f * 20.0f) { losOffset = diff; }
588  float rotation = MathUtils.VectorToAngle(losOffset);
589 
590  Vector2 scale = new Vector2(
591  MathHelper.Clamp(losOffset.Length() / 256.0f, 4.0f, 5.0f), 3.0f);
592 
593  spriteBatch.Begin(SpriteSortMode.Deferred, transformMatrix: cam.Transform * Matrix.CreateScale(new Vector3(GameSettings.CurrentConfig.Graphics.LightMapScale, GameSettings.CurrentConfig.Graphics.LightMapScale, 1.0f)));
594  spriteBatch.Draw(visionCircle, new Vector2(ViewTarget.WorldPosition.X, -ViewTarget.WorldPosition.Y), null, Color.White, rotation,
595  new Vector2(visionCircle.Width * 0.2f, visionCircle.Height / 2), scale, SpriteEffects.None, 0.0f);
596  spriteBatch.End();
597  }
598  else
599  {
600  graphics.Clear(Color.White);
601  }
602 
603 
604  //--------------------------------------
605 
606  if (LosEnabled && LosMode != LosMode.None && ViewTarget != null)
607  {
608  Vector2 pos = ViewTarget.DrawPosition;
609 
610  Rectangle camView = new Rectangle(cam.WorldView.X, cam.WorldView.Y - cam.WorldView.Height, cam.WorldView.Width, cam.WorldView.Height);
611 
612  Matrix shadowTransform = cam.ShaderTransform
613  * Matrix.CreateOrthographic(GameMain.GraphicsWidth, GameMain.GraphicsHeight, -1, 1) * 0.5f;
614 
615  var convexHulls = ConvexHull.GetHullsInRange(ViewTarget.Position, cam.WorldView.Width*0.75f, ViewTarget.Submarine);
