20  private static void _hookLuaCsPatch(MethodBase __originalMethod, object[] __args, object __instance, out object result, HookMethodType hookType)
21  {
22  result = null;
23 
24  try
25  {
26  var funcAddr = ((long)__originalMethod.MethodHandle.GetFunctionPointer());
27  HashSet<(string, LuaCsCompatPatchFunc, ACsMod)> methodSet = null;
28  switch (hookType)
29  {
30  case HookMethodType.Before:
31  instance.compatHookPrefixMethods.TryGetValue(funcAddr, out methodSet);
32  break;
33  case HookMethodType.After:
34  instance.compatHookPostfixMethods.TryGetValue(funcAddr, out methodSet);
35  break;
36  default:
37  throw new ArgumentException($"Invalid {nameof(HookMethodType)} enum value.", nameof(hookType));
38  }
39 
40  if (methodSet != null)
41  {
42  var @params = __originalMethod.GetParameters();
43  var args = new Dictionary<string, object>();
44  for (int i = 0; i < @params.Length; i++)
45  {
46  args.Add(@params[i].Name, __args[i]);
47  }
48 
49  var outOfSocpe = new HashSet<(string, LuaCsCompatPatchFunc, ACsMod)>();
50  foreach (var tuple in methodSet)
51  {
52  if (tuple.Item3 != null && tuple.Item3.IsDisposed)
53  {
54  outOfSocpe.Add(tuple);
55  }
56  else
57  {
58  var _result = tuple.Item2(__instance, args);
59  if (_result != null)
60  {
61  if (_result is DynValue res)
62  {
63  if (!res.IsNil())
64  {
65  if (__originalMethod is MethodInfo mi && mi.ReturnType != typeof(void))
66  {
67  result = res.ToObject(mi.ReturnType);
68  }
69  else
70  {
71  result = res.ToObject();
72  }
73  }
74  }
75  else
76  {
77  result = _result;
78  }
79  }
80  }
81  }
82  foreach (var tuple in outOfSocpe) { methodSet.Remove(tuple); }
83  }
84  }
85  catch (Exception ex)
86  {
87  GameMain.LuaCs.PrintError($"Error in {__originalMethod.Name}:", LuaCsMessageOrigin.Unknown);
88  GameMain.LuaCs.HandleException(ex, LuaCsMessageOrigin.Unknown);
89  }
90  }
