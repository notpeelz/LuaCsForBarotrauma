155  private void DefaultExceptionHandler(Exception ex, LuaCsMessageOrigin origin)
156  {
157  switch (ex)
158  {
159  case NetRuntimeException netRuntimeException:
160  if (netRuntimeException.DecoratedMessage == null)
161  {
162  PrintError(netRuntimeException, origin);
163  }
164  else
165  {
166  // FIXME: netRuntimeException.ToString() doesn't print the InnerException's stack trace...
167  PrintError($"{netRuntimeException.DecoratedMessage}: {netRuntimeException}", origin);
168  }
169  break;
170  case InterpreterException interpreterException:
171  if (interpreterException.DecoratedMessage == null)
172  {
173  PrintError(interpreterException, origin);
174  }
175  else
176  {
177  PrintError(interpreterException.DecoratedMessage, origin);
178  }
179  break;
180  default:
181  var msg = ex.StackTrace != null
182  ? ex.ToString()
183  : $"{ex}\n{Environment.StackTrace}";
184  PrintError(msg, origin);
185  break;
186  }
187  }

202  for (int i = 0; i < str.Length; i += 1024)
