182  public void HttpRequest(string url, LuaCsAction callback, string data = null, string method = "POST", string contentType = "application/json")
183  {
184  try
185  {
186  var httpWebRequest = (HttpWebRequest)WebRequest.Create(url);
187  httpWebRequest.ContentType = contentType;
188  httpWebRequest.Method = method;
189 
190  if (data != null)
191  {
192  using (var streamWriter = new StreamWriter(httpWebRequest.GetRequestStream()))
193  streamWriter.Write(data);
194  }
195 
196  httpWebRequest.BeginGetResponse(new AsyncCallback((IAsyncResult result) =>
197  {
198  try
199  {
200  var httpResponse = httpWebRequest.EndGetResponse(result);
201  using (var streamReader = new StreamReader(httpResponse.GetResponseStream()))
202  {
203  string responseResult = streamReader.ReadToEnd();
204  GameMain.LuaCs.Timer.Wait((object[] par) => { callback(responseResult); }, 0);
205  }
206  }
207  catch (Exception e)
208  {
209  GameMain.LuaCs.Timer.Wait((object[] par) => { callback(e.Message); }, 0);
210  }
211  }), null);
212 
213  }
214  catch (Exception e)
215  {
216  GameMain.LuaCs.Timer.Wait((object[] par) => { callback(e.Message); }, 0);
217  }
