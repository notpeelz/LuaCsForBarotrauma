71  private static void SpawnStartItems(Submarine sub, Identifier? startItemSet)
72  {

74  if (!StartItemSet.Sets.TryGet(setIdentifier, out StartItemSet itemSet))
75  {
76  DebugConsole.AddWarning($"Couldn't find a start item set matching the identifier \"{setIdentifier}\"!");
77  if (!StartItemSet.Sets.TryGet(DefaultStartItemSet, out StartItemSet defaultSet))
78  {
79  DebugConsole.ThrowError($"Couldn't find the default start item set \"{DefaultStartItemSet}\"!");
80  return;
81  }
82  itemSet = defaultSet;
83  }
84  WayPoint wp = WayPoint.GetRandom(SpawnType.Cargo, null, sub);
85  ISpatialEntity initialSpawnPos;
86  if (wp?.CurrentHull == null)
87  {
88  var spawnHull = Hull.HullList.Where(h => h.Submarine == sub && !h.IsWetRoom).GetRandomUnsynced();
89  if (spawnHull == null)
90  {
91  DebugConsole.AddWarning($"Failed to spawn start items in the sub. No cargo waypoint or dry hulls found to spawn the items in.");
92  return;
93  }
94  initialSpawnPos = spawnHull;
95  }
96  else
97  {
98  initialSpawnPos = wp;
99  }
100  var newItems = new List<Item>();
101  foreach (var startItem in itemSet.Items)
102  {
103  if (!ItemPrefab.Prefabs.TryGet(startItem.Item, out ItemPrefab itemPrefab))
104  {
105  DebugConsole.AddWarning($"Cannot find a start item with with the identifier \"{startItem.Item}\"");
106  continue;
107  }
108  for (int i = 0; i < startItem.Amount; i++)
109  {
110  var item = new Item(itemPrefab, initialSpawnPos.Position, sub, callOnItemLoaded: false);
111  // Is this necessary?
112  foreach (WifiComponent wifiComponent in item.GetComponents<WifiComponent>())
113  {
114  wifiComponent.TeamID = sub.TeamID;
115  }
116  newItems.Add(item);
117  }
118  }
119  var cargoContainers = new List<ItemContainer>();
120  foreach (var item in newItems)
121  {
122 #if SERVER
123  Entity.Spawner.CreateNetworkEvent(new EntitySpawner.SpawnEntity(item));
124 #endif
125  foreach (ItemComponent ic in item.Components)
126  {
127  ic.OnItemLoaded();
128  }
129  var container = sub.FindContainerFor(item, onlyPrimary: true);
130  if (container == null)
131  {
132  var cargoContainer = CargoManager.GetOrCreateCargoContainerFor(item.Prefab, initialSpawnPos, ref cargoContainers);
133  container = cargoContainer?.Item;

140  {

147  List<Item> itemsToSpawn = new List<Item>(100);
148 
149  int itemCountApprox = MapEntityPrefab.List.Count() / 3;
150  var containers = new List<ItemContainer>(70 + 30 * subs.Count());
151  var prefabsItemsCanSpawnIn = new List<ItemPrefab>(itemCountApprox / 3);
152  var singlePrefabs = new List<ItemPrefab>(itemCountApprox);
153  var removals = new List<ItemPrefab>();

155  // generate loot only for a specific container if defined

185  bool CanSpawnIn(ItemPrefab item, ItemPrefab container)
186  {
187  foreach (var preferredContainer in item.PreferredContainers)
188  {
189  if (ItemPrefab.IsContainerPreferred(preferredContainer.Primary, container.Identifier.ToEnumerable().Union(container.Tags))) { return true; }
190  }
191  return false;
192  }
193 
194  var validContainers = new Dictionary<ItemContainer, PreferredContainer>();
195  prefabsItemsCanSpawnIn.Shuffle(Rand.RandSync.ServerAndClient);
196  // Spawn items that other items can spawn in first so we can fill them up with items if needed (oxygen tanks inside the spawned diving masks, etc)
197  for (int i = 0; i < prefabsItemsCanSpawnIn.Count; i++)
198  {
199  var itemPrefab = prefabsItemsCanSpawnIn[i];
200  if (itemPrefab == null) { continue; }

203 
204  // Spawn items that nothing can spawn in last
205  singlePrefabs.Shuffle(Rand.RandSync.ServerAndClient);

207 
208  if (OutputDebugInfo)
209  {

212  foreach (string itemName in itemsToSpawn.Select(it => it.Name).Distinct())
213  {

215  }
216  }
217

219  GameMain.GameSession.Level.Type == LevelData.LevelType.Outpost &&

221  {

231  matchingItem.Remove();
232  itemsToSpawn.Remove(matchingItem);
233  foreach (Item containedItem in containedItems)
234  {
235  containedItem.Remove();
236  itemsToSpawn.Remove(containedItem);

238  }
239  }

251  bool SpawnItems(ItemPrefab itemPrefab)
252  {
253  if (itemPrefab == null)
254  {
255  string errorMsg = "Error in AutoItemPlacer.SpawnItems - itemPrefab was null.\n" + Environment.StackTrace.CleanupStackTrace();
256  DebugConsole.ThrowError(errorMsg);
257  GameAnalyticsManager.AddErrorEventOnce("AutoItemPlacer.SpawnItems:ItemNull", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
258  return false;
259  }
260  bool success = false;
261  bool isCampaign = GameMain.GameSession?.GameMode is CampaignMode;
262  foreach (PreferredContainer preferredContainer in itemPrefab.PreferredContainers)
263  {
264  if (preferredContainer.CampaignOnly && !isCampaign) { continue; }
265  if (preferredContainer.NotCampaign && isCampaign) { continue; }
266  if (preferredContainer.SpawnProbability <= 0.0f || preferredContainer.MaxAmount <= 0 && preferredContainer.Amount <= 0) { continue; }
267  validContainers = GetValidContainers(preferredContainer, containers, validContainers, primary: true);
268  if (validContainers.None())
269  {
270  validContainers = GetValidContainers(preferredContainer, containers, validContainers, primary: false);
271  }
272  foreach (var validContainer in validContainers)
273  {
274  var newItems = CreateItems(itemPrefab, containers, validContainer);
275  if (newItems.Any())
276  {
277  itemsToSpawn.AddRange(newItems);
278  success = true;
279  }
280  }
281  }
282  return success;
283  }

286  private static Dictionary<ItemContainer, PreferredContainer> GetValidContainers(PreferredContainer preferredContainer, IEnumerable<ItemContainer> allContainers, Dictionary<ItemContainer, PreferredContainer> validContainers, bool primary)
287  {
288  validContainers.Clear();
289  foreach (ItemContainer container in allContainers)
290  {
291  if (!container.AutoFill) { continue; }
292  if (primary)
293  {
294  if (!ItemPrefab.IsContainerPreferred(preferredContainer.Primary, container)) { continue; }
295  }
296  else
297  {
298  if (!ItemPrefab.IsContainerPreferred(preferredContainer.Secondary, container)) { continue; }
299  }
300  if (!validContainers.ContainsKey(container))
301  {
302  validContainers.Add(container, preferredContainer);
303  }
304  }
305  return validContainers;
306  }

316  private static List<Item> CreateItems(ItemPrefab itemPrefab, List<ItemContainer> containers, KeyValuePair<ItemContainer, PreferredContainer> validContainer)
317  {
318  List<Item> newItems = new List<Item>();
319  if (Rand.Value(Rand.RandSync.ServerAndClient) > validContainer.Value.SpawnProbability) { return newItems; }
320  // Don't add dangerously reactive materials in thalamus wrecks
321  if (validContainer.Key.Item.Submarine.WreckAI != null && itemPrefab.Tags.Contains("explodesinwater"))
322  {
323  return newItems;
324  }
325  int amount = validContainer.Value.Amount;
326  if (amount == 0)
327  {
328  amount = Rand.Range(validContainer.Value.MinAmount, validContainer.Value.MaxAmount + 1, Rand.RandSync.ServerAndClient);
329  }
330  for (int i = 0; i < amount; i++)
331  {
332  if (validContainer.Key.Inventory.IsFull(takeStacksIntoAccount: true))
333  {
334  containers.Remove(validContainer.Key);
335  break;
336  }
337  var existingItem = validContainer.Key.Inventory.AllItems.FirstOrDefault(it => it.Prefab == itemPrefab);
338  int quality =
339  existingItem?.Quality ??
340  ToolBox.SelectWeightedRandom(
341  qualityCommonnesses.Select(q => q.quality).ToList(),
342  qualityCommonnesses.Select(q => q.commonness).ToList(), Rand.RandSync.ServerAndClient);
343  if (!validContainer.Key.Inventory.CanBePut(itemPrefab, quality: quality)) { break; }
344  var item = new Item(itemPrefab, validContainer.Key.Item.Position, validContainer.Key.Item.Submarine, callOnItemLoaded: false)
345  {
346  SpawnedInCurrentOutpost = validContainer.Key.Item.SpawnedInCurrentOutpost,
347  AllowStealing = validContainer.Key.Item.AllowStealing,
348  Quality = quality,
349  OriginalModuleIndex = validContainer.Key.Item.OriginalModuleIndex,
350  OriginalContainerIndex =
351  Item.ItemList.Where(it => it.Submarine == validContainer.Key.Item.Submarine && it.OriginalModuleIndex == validContainer.Key.Item.OriginalModuleIndex).ToList().IndexOf(validContainer.Key.Item)
352  };
353  foreach (WifiComponent wifiComponent in item.GetComponents<WifiComponent>())
354  {
355  wifiComponent.TeamID = validContainer.Key.Item.Submarine.TeamID;
356  }
357  newItems.Add(item);
358  validContainer.Key.Inventory.TryPutItem(item, null, createNetworkEvent: false);
359  containers.AddRange(item.GetComponents<ItemContainer>());
360  }
361  return newItems;
362  }
