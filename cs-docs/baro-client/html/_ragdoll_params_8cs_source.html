130  public static T GetRagdollParams<T>(Identifier speciesName, string fileName = null) where T : RagdollParams, new()
131  {
132  if (speciesName.IsEmpty)
133  {
134  throw new Exception($"Species name null or empty!");
135  }
136  if (!allRagdolls.TryGetValue(speciesName, out Dictionary<string, RagdollParams> ragdolls))
137  {
138  ragdolls = new Dictionary<string, RagdollParams>();
139  allRagdolls.Add(speciesName, ragdolls);
140  }
141  if (!string.IsNullOrEmpty(fileName) && ragdolls.TryGetValue(fileName, out RagdollParams ragdoll))
142  {
143  return (T)ragdoll;
144  }
145  string selectedFile = null;
146  Identifier ragdollSpecies = speciesName;
147  if (CharacterPrefab.Prefabs.TryGet(speciesName, out var prefab))
148  {
149  if (!prefab.VariantOf.IsEmpty)
150  {
151  ragdollSpecies = prefab.VariantOf;
152  }
153  string error = null;
154  string folder = GetFolder(ragdollSpecies);
155  if (!Directory.Exists(folder))
156  {
157  error = $"[RagdollParams] Invalid directory: {folder}. Using the default ragdoll.";
158  selectedFile = GetDefaultFile(ragdollSpecies);
159  }
160  else
161  {
162  string[] files = Directory.GetFiles(folder);
163  if (files.None())
164  {
165  error = $"[RagdollParams] Could not find any ragdoll files from the folder: {folder}. Using the default ragdoll.";
166  selectedFile = GetDefaultFile(ragdollSpecies);
167  }
168  else if (string.IsNullOrEmpty(fileName))
169  {
170  // Files found, but none specified
171  selectedFile = GetDefaultFile(ragdollSpecies);
172  }
173  else
174  {
175  selectedFile = files.FirstOrDefault(f => IO.Path.GetFileNameWithoutExtension(f).Equals(fileName, StringComparison.OrdinalIgnoreCase));
176  if (selectedFile == null)
177  {
178  error = $"[RagdollParams] Could not find a ragdoll file that matches the name {fileName}. Using the default ragdoll.";
179  selectedFile = GetDefaultFile(ragdollSpecies);
180  }
181  }
182  }
183  if (error != null)
184  {
185  DebugConsole.ThrowError(error);
186  }
187  }
188  if (selectedFile == null)
189  {
190  throw new Exception("[RagdollParams] Selected file null!");
191  }
192  DebugConsole.Log($"[RagdollParams] Loading ragdoll from {selectedFile}.");
193  var characterPrefab = CharacterPrefab.Prefabs[speciesName];
194  T r = new T();
195  if (r.Load(ContentPath.FromRaw(characterPrefab.ContentPackage, selectedFile), ragdollSpecies))
196  {
197  if (!ragdolls.ContainsKey(r.Name))
198  {
199  ragdolls.Add(r.Name, r);
200  }
201  return r;
202  }
203  else
204  {
205  // Failing to create a ragdoll causes so many issues that cannot be handled. Dummy ragdoll just seems to make things harded to debug. It's better to fail early.
206  throw new Exception($"[RagdollParams] Failed to load ragdoll {r.Name} from {selectedFile} for the character {speciesName}.");
207  }
208  }

436  private void RevertTo(RagdollParams source)
437  {
438  if (source.MainElement == null)
439  {
440  DebugConsole.ThrowError("[RagdollParams] The source XML Element of the given RagdollParams is null!");
441  return;
442  }
443  Deserialize(source.MainElement, alsoChildren: false);
444  var sourceSubParams = source.GetAllSubParams().ToList();
445  var subParams = GetAllSubParams().ToList();
446  // TODO: cannot currently undo joint/limb deletion.
447  if (sourceSubParams.Count != subParams.Count)
448  {
449  DebugConsole.ThrowError("[RagdollParams] The count of the sub params differs! Failed to revert to the previous snapshot! Please reset the ragdoll to undo the changes.");
450  return;
451  }
452  for (int i = 0; i < subParams.Count; i++)
453  {
454  var subSubParams = subParams[i].SubParams;
455  if (subSubParams.Count != sourceSubParams[i].SubParams.Count)
456  {
457  DebugConsole.ThrowError("[RagdollParams] The count of the sub sub params differs! Failed to revert to the previous snapshot! Please reset the ragdoll to undo the changes.");
458  return;
459  }
460  subParams[i].Deserialize(sourceSubParams[i].Element, recursive: false);
461  for (int j = 0; j < subSubParams.Count; j++)
462  {
463  subSubParams[j].Deserialize(sourceSubParams[i].SubParams[j].Element, recursive: false);
464  // Since we cannot use recursion here, we have to go deeper manually, if necessary.
465  }
466  }
